Sudo Slider jQuery Plugin - Slide on event (ajax) demo

Usage

Include the javascripts

<script type="Text/Javascript" src="http://code.jquery.com/jquery-1.11.0.min.js"></script>
<script type="text/javascript" src="../js/jquery.sudoSlider.min.js"></script>

The Javascript to start it.

<script type="text/javascript" >
   $(document).ready(function(){	
      // The sudoSlider variable must be gloabal for the javascript in the document form.html to work. 
      sudoSlider = $("#slider").sudoSlider({ 
         prevNext: false,
         ajax: ['AjaxDependencies/form.html', 'AjaxDependencies/succes.html'],
         customLink:'a.eventlink'
      });
   });
</script>
